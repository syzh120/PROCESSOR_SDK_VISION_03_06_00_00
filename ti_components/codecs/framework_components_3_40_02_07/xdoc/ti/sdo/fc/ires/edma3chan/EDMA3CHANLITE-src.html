    33    /*!
    34     *  ======== EDMA3CHANLITE ========
    35     *  EDMA3-specific IRES Resman and Protocol Implementation
    36     *
    37     */
    38    
    39    @Template("./EDMA3CHANLITE.xdt")
    40    
    41    metaonly module EDMA3CHANLITE inherits ti.sdo.fc.ires.IRESMAN {
    42    
    43        /* Maximum number of TCCS that can be requested in a single logical handle*/
    44        const Int MAX_TCCS = 3;
    45    
    46        /* Maximum number of PARAMS that can be requested in single logical handle*/
    47        const Int MAX_PARAMS = 5;
    48    
    49        /* Maximum number of logical handles that can be requested */
    50        const Int MAX_RESOURCES = 5;
    51    
    52        /* Some way to represent a pool of resources */
    53        struct ResourcePool{
    54            UInt chan;
    55            UInt numTccs;
    56            UInt tccArray[MAX_TCCS];
    57            UInt numParams;
    58            UInt paramArray[MAX_PARAMS];
    59        };
    60    
    61        /* Total number of channels available */
    62        config Int numLogicalChans = 0;
    63    
    64        /*!
    65         *  ======== baseAddress ========
    66         *  Base address of the EDMA3 Channel Controller
    67         */
    68        config UInt baseAddress;
    69    
    70        /* Each element of array is one logical channel */
    71        config ResourcePool logicalChans[MAX_RESOURCES];
    72    }
