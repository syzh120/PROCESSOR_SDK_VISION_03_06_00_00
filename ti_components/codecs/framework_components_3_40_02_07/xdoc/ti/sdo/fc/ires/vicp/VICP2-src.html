    34    /*!
    35     *  ======== VICP2 ========
    36     *  VICP2-specific IRES Resman and Protocol Implementation
    37     *
    38     */
    39    @Template("./VICP2.xdt")
    40    
    41    metaonly module VICP2 inherits ti.sdo.fc.ires.IRESMAN {
    42    
    43        /*!
    44         *  ======== ipcKeyBase ========
    45         *  Linux only. Base value of keys of Linux IPC objects used by
    46         *  VICP2. The IPC objects created by VICP2 will use keys starting
    47         *  at this
    48         *  value, and incrementing with each new object. There are currently
    49         *  three IPC objects, so keys in the range of ipcKeyBase to ipcKeyBase + 2
    50         *  will be reserved for VICP2. The default value of ipcKeyBase is
    51         *  ascii code for "PCIV".
    52         *
    53         *  WARNING: This value should only be changed if it conflicts with
    54         *  another IPC key in the system that cannot be changed. If this value
    55         *  is changed, all programs using VICP2 that will be run simultaneously
    56         *  must have the ipcKeyBase configured to the new value.
    57         *  @_nodoc
    58         */
    59        config UInt ipcKeyBase = 0x50434956;
    60    
    61        /*  @_nodoc */
    62        override config Bool autoRegister = true;
    63    }
