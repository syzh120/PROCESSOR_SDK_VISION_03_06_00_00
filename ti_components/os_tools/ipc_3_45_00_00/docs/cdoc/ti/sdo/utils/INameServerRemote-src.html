    37    import xdc.runtime.Error;
    38    
    39    /*!
    40     *  ======== INameServerRemote ========
    41     *  Interface for remote NameServer accesses
    42     */
    43    interface INameServerRemote {
    44    
    45        /*!
    46         *  ======== Status ========
    47         *  Status codes returned by certain calls
    48         */
    49        enum Status {
    50            S_SUCCESS = 0,
    51            E_FAIL = -1,
    52            E_ERROR = -2
    53        };
    54    
    55        /*! Function is called by Ipc_attach() through NameServer_SetupProxy */
    56        @DirectCall
    57        Int attach(UInt16 remoteProcId, Ptr sharedAddr);
    58    
    59        /*! Function is called by Ipc_detach() through NameServer_SetupProxy */
    60        @DirectCall
    61        Int detach(UInt16 remoteProcId);
    62    
    63        /*!
    64         *  ======== sharedMemReq ========
    65         *  Returns the shared memory size requirement for a single instance
    66         *
    67         *  Function is called during Ipc_attach() through NameServer_SetupProxy.
    68         *
    69         *  @param(sharedAddr)   Shared address
    70         *
    71         *  @b(returns)          Number of MAUs needed to create the instance.
    72         */
    73        @DirectCall
    74        SizeT sharedMemReq(Ptr sharedAddr);
    75    
    76    instance:
    77    
    78        /*!
    79         *  ======== create ========
    80         */
    81        override create(UInt16 procId);
    82    
    83        /*!
    84         *  ======== get ========
    85         */
    86        Int get(String instanceName,
    87                String name,
    88                Ptr value,
    89                UInt32 *valueLen,
    90                xdc.runtime.knl.ISync.Handle syncHandle,
    91                Error.Block *eb);
    92    }
