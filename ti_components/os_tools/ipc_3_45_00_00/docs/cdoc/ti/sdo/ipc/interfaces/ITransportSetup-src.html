    42    interface ITransportSetup
    43    {
    44        /*! priority of the transport */
    45        config UInt priority = 0;
    46    
    47        /*! Function that will be called in MessageQ_attach */
    48        @DirectCall
    49        Int attach(UInt16 remoteProcId, Ptr sharedAddr);
    50    
    51        /*! Function that will be called in MessageQ_detach */
    52        @DirectCall
    53        Int detach(UInt16 remoteProcId);
    54    
    55        /*! Determines if a transport has been registered to a remote processor */
    56        @DirectCall
    57        Bool isRegistered(UInt16 remoteProcId);
    58    
    59        /*! Shared memory required */
    60        @DirectCall
    61        SizeT sharedMemReq(Ptr sharedAddr);
    62    }
