    37    import xdc.runtime.Assert;
    38    
    39    /*!
    40     *  ======== GateMPSupportNull ========
    41     *  Module to plug GateMP proxies
    42     */
    43    module GateMPSupportNull inherits ti.sdo.ipc.interfaces.IGateMPSupport
    44    {
    45    
    46        /*!
    47         *  Assert raised when trying to use GateMPSupportNull's enter or leave
    48         */
    49        config Assert.Id A_invalidAction  = {
    50            msg: "A_invalidAction: Cannot use ti.sdo.ipc.gates.GateMPSupportNull"
    51        };
    52    
    53        /*!
    54         *  Error codes returned by certain calls in GateMP
    55         */
    56        enum Action {
    57            Action_NONE   =  0,
    58            Action_ASSERT =  1
    59        };
    60    
    61        /*!
    62         *  ======== action ========
    63         *  Assert if the enter and/or leave is called
    64         */
    65        config Action action = Action_ASSERT;
    66    
    67    instance:
    68    
    69        /*!
    70         *  @_nodoc
    71         *  ======== enter ========
    72         *  Enter this gate
    73         */
    74        @DirectCall
    75        override IArg enter();
    76    
    77        /*!
    78         *  @_nodoc
    79         *  ======== leave ========
    80         *  Leave this gate
    81         */
    82        @DirectCall
    83        override Void leave(IArg key);
    84    
    85    internal:
    86    
    87        struct Instance_State {
    88            UInt   resourceId;
    89        };
    90    
    91    }
