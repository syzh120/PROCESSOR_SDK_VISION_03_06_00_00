    42     */

    44    {

    46        config UInt dspIntVectId = 5;

    48        /*  Setup notify driver with host.
    49         *

    51         *  be included in the IPC attach phase which will ultimately
    52         *  create an instance of the notify circ driver. When set to
    53         *  false, the host is excluded.
    54         */

    58    
    59    }
